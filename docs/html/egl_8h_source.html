1307static int sf_glad_egl_find_core_egl(EGLDisplay display) {
1308 int major, minor;
1309 const char *version;
1310
1311 if (display == NULL) {
1312 display = EGL_NO_DISPLAY; /* this is usually NULL, better safe than sorry */
1313 }
1314 if (display == EGL_NO_DISPLAY) {
1315 display = eglGetCurrentDisplay();
1316 }
1317#ifdef EGL_VERSION_1_4
1318 if (display == EGL_NO_DISPLAY) {
1319 display = eglGetDisplay(EGL_DEFAULT_DISPLAY);
1320 }
1321#endif
1322#ifndef EGL_VERSION_1_5
1323 if (display == EGL_NO_DISPLAY) {
1324 return 0;
1325 }
1326#endif
1327
1328 version = eglQueryString(display, EGL_VERSION);
1329 (void) eglGetError();
1330
1331 if (version == NULL) {
1332 major = 1;
1333 minor = 0;
1334 } else {
1335 GLAD_IMPL_UTIL_SSCANF(version, "%d.%d", &major, &minor);
1336 }
1337
1338 SF_GLAD_EGL_VERSION_1_0 = (major == 1 && minor >= 0) || major > 1;
1339 SF_GLAD_EGL_VERSION_1_1 = (major == 1 && minor >= 1) || major > 1;
1340 SF_GLAD_EGL_VERSION_1_2 = (major == 1 && minor >= 2) || major > 1;
1341 SF_GLAD_EGL_VERSION_1_3 = (major == 1 && minor >= 3) || major > 1;
1342 SF_GLAD_EGL_VERSION_1_4 = (major == 1 && minor >= 4) || major > 1;
1343 SF_GLAD_EGL_VERSION_1_5 = (major == 1 && minor >= 5) || major > 1;
1344
1345 return GLAD_MAKE_VERSION(major, minor);
1346}
